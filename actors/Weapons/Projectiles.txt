Actor BFLStandardShot
{
  PROJECTILE
  DONTHURTSHOOTER
  Radius 5
  Height 5
  Speed 250
  Damage 100
  +DoomBounce
  +Ripper
  +ForceRadiusDmg
  Renderstyle Add
  BounceCount 4
  Scale 0.50
  Alpha 0.95
  Obituary "%o was vaporized by %k's Big Frickin Laser."
  Decal BigScorch
  States
  {
  Spawn:
    BFLS A 0 A_SpawnItemEx("StandardBFLTrail", -5)
    BFLS A 1 Bright A_SpawnItemEx("StandardBFLTrail", -5)
    Goto DmgLoop
  DmgLoop:
    BFLS A 0 A_SpawnItemEx("StandardBFLTrail", -5)
    BFLS A 1 Bright A_Explode(100,100,0,0,0,0,0,BulletPuff,"BFLBeam")
    BFLS A 0 A_SpawnItemEx("StandardBFLTrail", -5)
    Loop
  Death:
    BFLS BCDEFG 2 Bright
    Stop
   }
}
